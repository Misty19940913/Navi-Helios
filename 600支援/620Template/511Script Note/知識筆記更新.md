<%*
const folder = "03 知識庫";   // 你的知識庫資料夾
const files = app.vault.getMarkdownFiles();

const template = `---
type: knowledge
aliases:
tags:
level:
created: <% tp.date.now() %>
updated: <% tp.date.now() %>
used_in: []
---

## 核心

## 情境

## 演化

## 反向整理
- 使用於專案：
- 解決的問題：
- 實作或輸出：

## 深入
- 相關筆記：
- 高階概念：
`;

for (const f of files) {
    if (!f.path.contains(folder)) continue;

    const content = await app.vault.read(f);

    // 若已升級過則跳過
    if (content.startsWith("---") && content.contains("type: knowledge")) continue;

    // 補上 template 到最前
    const newContent = template + "\n" + content;
    await app.vault.modify(f, newContent);
}

tR += "所有知識卡片已升級完畢。";
%>
